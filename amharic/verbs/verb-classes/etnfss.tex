%
%  This is an Ethiopic character mapping file for Mule users who want to use
%  the "ethiop" fonts WITHOUT using Babel.  To set up the fonts and use the
%  LaTeX 2e NFSS system, something similar to the following should work for you:
%
%  1)  ftp://ftp.dante.de/tex-archive/languages/ethiopia/ethiop/
%      obtain all files under "latex", "src", and "tfm".
%
%  2)  mv the "src" and "tfm" directories to /usr/lib/texmf/fonts/ethiop/
%      or to wherever you keep tex fonts  -you may use them locally under
%      the directory where you will compile your document (...in theory).
%
%  3)  in the "latex" directory where you have downloaded "ethiop.ins" and
%      "ethiop.dtx" you will need to execute "latex ethiop.ins"
%
%  4)  mv the "uetha.fd" and "uethb.fd" to /usr/lib/texmf/tex/latex/fd/
%      or your local equivalent :-)  It is advicable to change the "u" of
%      "ueth[ab].tex" to the uppercase "U", or make a link accordingly.
%
%  5)  Test, compile the file "fidel.tex" (from where you found this file)
%      with "latex fidel.tex".  View with "xdvi fidel.dvi", or convert to PS
%      and print/view as you wish.
%
%      You may need to run xdvi twice, kpathsea which your system might use to
%      create new pk files, seems not to inform xdvi the first time that the
%      just created font is now available.
%
%      Good Luck!  Send comments/bugs to Daniel Yacob <dmulholl@cs.indiana.edu>
%
%

%
%  For details on obtaining and installing ethiop
%  visit the URL:  
%
%  http://www.informatik.uni-hamburg.de/TGI/mitarbeiter/wimis/kummer/ethiop_eng.html
%  
%  or
%
%  ftp://ftp.dante.de/tex-archive/languages/ethiopia/ethiop/
%
%  To use the ethiop fonts with Babel and Mule as originally intended, see 
%  "etbabel.tex"  under  ftp://ftp.cs.indiana.edu/pub/fidel/latex/
%


% 
%  This file may be used in LaTeX as per \input{etnfss} when used also with
%  the ethiop babel package and the Mule 2.3+  and Emacs 20.1+ editors.  Since
%  the metafonts provide a syllabary somewhat smaller than the proposed Extended
%  Ethiopic Unicode specification of Mule;  some hacks are added here so no
%  characters are lost.  The results may not be satisfactory.  A revised metafont,
%  to meet the Unicode specification, will likely come sometime in the future.
%  There will NOT be a compatibility problem between systems when the escapes
%  sequences in this map file are used.
%

%
%  Borrowed from ipamacs' \diatop
%
\def\diatopx[#1|#2]{{\setbox1=\hbox{{#1{}}}\setbox2=\hbox{{#2{}}}%
                    \dimen0=\ifdim\wd1>\wd2\wd1\else\wd2\fi%
                    \dimen1=\ht2\advance\dimen1by-1ex%
%                   \dimen1=\ht2\advance\dimen1by+0.05ex%
                    \setbox1=\hbox to1\dimen0{\hss#1\hss}%
                    \leavevmode % added by D. Roegel
                    \rlap{\raise1\dimen1\box1}%
                    \hbox to1\dimen0{\hss#2\hss}}}%

%  The below macros are used to create a few extra series missing from the
%  Ethiop Metafonts  ...though not very well.

\def\AddWa#1{{\setbox1=\hbox{{#1{}}}\setbox2=\hbox{{\WaG{}}}%
                    \dimen0=\ifdim\wd1>\wd2\wd1\else\wd2\fi%
                    \dimen1=\ht2\advance\dimen1by-1.35ex%
                    \setbox1=\hbox to1\dimen0{\hss#1\hss}%
                    \leavevmode % added by D. Roegel
                    \rlap{\raise1\dimen1\box1}%
                    \hbox to1\dimen0{\hss{\WaG}\hss}}}%
